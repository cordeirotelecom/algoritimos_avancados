// Algoritmo de Dijkstra
// Complexidade: O((V + E) log V) com heap
// Usa: Encontrar caminho mais curto em grafos ponderados com pesos positivos

class GrafoComPeso {
    constructor() {
        this.adjacencia = {};
    }
    
    adicionarVertice(vertice) {
        if (!this.adjacencia[vertice]) {
            this.adjacencia[vertice] = [];
        }
    }
    
    adicionarAresta(v1, v2, peso) {
        this.adjacencia[v1].push({ vertice: v2, peso });
        this.adjacencia[v2].push({ vertice: v1, peso });
    }
    
    dijkstra(origem) {
        const distancias = {};
        const predecessores = {};
        const visitados = new Set();
        const vertices = Object.keys(this.adjacencia);
        
        // Inicializar dist√¢ncias
        vertices.forEach(v => {
            distancias[v] = Infinity;
            predecessores[v] = null;
        });
        distancias[origem] = 0;
        
        // Processar v√©rtices
        while (visitados.size < vertices.length) {
            // Encontrar v√©rtice n√£o visitado com menor dist√¢ncia
            let menorDistancia = Infinity;
            let verticeAtual = null;
            
            vertices.forEach(v => {
                if (!visitados.has(v) && distancias[v] < menorDistancia) {
                    menorDistancia = distancias[v];
                    verticeAtual = v;
                }
            });
            
            if (verticeAtual === null) break;
            
            visitados.add(verticeAtual);
            
            // Atualizar dist√¢ncias dos vizinhos
            this.adjacencia[verticeAtual].forEach(({ vertice, peso }) => {
                if (!visitados.has(vertice)) {
                    const novaDistancia = distancias[verticeAtual] + peso;
                    
                    if (novaDistancia < distancias[vertice]) {
                        distancias[vertice] = novaDistancia;
                        predecessores[vertice] = verticeAtual;
                    }
                }
            });
        }
        
        return { distancias, predecessores };
    }
    
    reconstruirCaminho(predecessores, destino) {
        const caminho = [];
        let atual = destino;
        
        while (atual !== null) {
            caminho.unshift(atual);
            atual = predecessores[atual];
        }
        
        return caminho;
    }
}
